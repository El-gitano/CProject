def initGrille(taille)

        @cases = Array.new
        
        0.upto(taille-1){|x|
        
                colonne = Array.new
                
                0.upto(taille-1){|y|
        
                        colonne.push(Case.new(x, y))
                }
                
                @cases.push(colonne)
        }
end
